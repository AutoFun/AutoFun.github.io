\hline e^{j \theta}=\cos \theta+j \sin \theta& \cos (\theta)=\frac{1}{2}\left(e^{j \theta}+e^{-j \theta}\right)&\sin (\theta)=\frac{1}{2^{i}}\left(e^{j \theta}-e^{-j \theta}\right)\
\hline
\end{array}

Trigonometric identities

$$
\begin{array}{|c|c|}
\hline \sin A+\sin B=2 \sin \frac{A+B}{2} \cos \frac{A-B}{2}&\sin A \cos B=\frac{1}{2}[\sin (A+B)+\sin (A-B)] \
\hline \sin A-\sin B=2 \sin \frac{A-B}{2} \cos \frac{A+B}{2}&\cos A \sin B=\frac{1}{2}[\sin (A+B)-\sin (A-B)] \
\hline \cos A+\cos B=2 \cos \frac{A+B}{2} \cos \frac{A-B}{2}&\cos A \cos B=\frac{1}{2}[\cos (A-B)+\cos (A+B)] \
\hline \cos A-\cos B=-2 \sin \frac{A+B}{2} \sin \frac{A-B}{2}&\sin A \sin B=\frac{1}{2}[\cos (A-B)-\cos (A+B)] \
\hline \sin ^{2} A=\frac{1}{2}[1-\cos 2 A]& cos ^{2} A=\frac{1}{2}[1+\cos 2 A]\
\hline
\end{array}
$$

Integral

$$
\begin{array}{|c|c|}
\hline \int x^{n} d x=\frac{1}{n+1} x^{n+1}+C & \
\hline \int \sin x d x=-\cos x+C & \int \cos x d x=\sin x+C \
\hline \int x \sin x d x=-x \cos x+\sin x+C & \int x^{2} \sin x d x=-x^{2} \cos x+2 x \sin x+2 \cos x+C \
\hline \int x \cos x d x=x \sin x+\cos x+C & \int x^{2} \cos x d x=x^{2} \sin x+2 x \cos x-2 \sin x+C \
\hline
\end{array}
$$

Sampling and aliasing

$$
\begin{array}{|c|c|}
\hline x(n)=x_{a}\left(n T_{s}\right) where T_{s} is the sampling period& x_{a}(t)=A \cos (2 \pi F t+\Phi)\
\hline & x(n)=A \cos \left(2 \pi f_{d} n+\Phi\right) \
\hline Normalised frequency:&f_{d}=F / F_{s} \
\hline & \omega_{d}=2 \pi f_{d}\
\hline Nyquist rate&=2 \times F_{\max } \
\hline Folding frequency& =\frac{F_{a}}{2} \
\hline Aliasing:&\cos (\theta)=\cos (\pm \theta+2 n \pi) \
\hline & F_{\text {alias }}=k \times F_{s} \pm f_{d} \times F_{s} \
\hline Signal-to-quantisation noise ratio:&\Delta=\frac{R}{L}, R=x_{\max }-x_{\min }, L=2^{B} \
\hline &P_{\text {noise }}=\frac{\Delta^{2}}{12} \
\hline & P_{\text {signal }}=\frac{R^{2}}{12}\
\hline & \text { SQNR }=6 \times B\
\hline
\end{array}
$$

Discrete-time Convolution

$$
\begin{array}{|c|c|}
\hline Definition:& \quad y(n)=x(n) * h(n)=\sum_{k=-\infty}^{\infty} x(k) h(n-k) \
\hline Commutative:& \quad x(n) * h(n)=h(n) * x(n) \
\hline Distributive:&\quad x(n) * h_{1}(n)+x(n) * h_{2}(n)=x(n) * \left[h_{1}(n)+h_{2}(n)\right] \
\hline Associative:& \left.\quad\left[x_{1}(n) * x_{2}(n)\right] * x_{3}(n)=x_{1}(n) * \left[x_{2}(n)\right) * x_{3}(n)\right]\
\hline With unit impulse:&x(n) * \delta\left(n-n_{0}\right)=x\left(n-n_{0}\right) \
\hline
\end{array}
$$

Random signals

$$
\begin{array}{|c|c|}
\hline CDF: & \quad P(x)= Probability [X \leq x] \
\hline PDF: & \quad p(x)=\frac{\partial P(x)}{\partial x} \quad or \quad P(x)=\int_{-\infty}^{x} p(u) d u\
\hline Properties: & p(x) \geq 0, for all x ; \quad \int_{-\infty}^{\infty} p(x) d x=1 \
\hline \quad & 0 \leq P(x) \leq 1 ; \quad P(-\infty)=0 ; \quad P(\infty)=1 ; P(x) is non-decreasing. \
\hline \quad & Probability\left[x_{1}<X \leq x_{2}\right]=P_{X}\left(x_{2}\right)-P_{X}\left(x_{1}\right)\
\hline Expected value: & \quad E[f(X)]=\int_{-\infty}^{\infty} f(x) p(x) d x \
\hline Mean: & \quad \mu_{X}=\int_{-\infty}^{\infty} x p(x) d x \quad\
\hline Mean-square:& \quad E\left[X^{2}\right]=\int_{-\infty}^{\infty} x^{2} p(x) d x\
\hline Variance: & \operatorname{var}[X]=E\left[X^{2}\right]-\mu_{X}^{2}\
\hline Standard deviation:& \sigma_{X}=\sqrt{\operatorname{var}[X]}
X uniformly distributed in [a, b]: E[X]=\frac{a+b}{2} and \operatorname{var}[X]=\frac{(b-a)^{2}}{12}\
\hline Independent:& p_{X Y}(x, y)=p_{X}(x) p_{Y}(y) \
\hline Uncorrelated:& E[X Y]=E[X] E[Y]\
\hline
\end{array}
$$

Methods for inverse-z transform

$$
\begin{array}{|l|c|c|}
\hline Power series: & X(z)=\sum_{n=0}^{\infty} \alpha_{n} z^{-n}=\alpha_{0}+\alpha_{1} z^{-1}+\alpha_{2} z^{-2}+\ldots+\alpha_{n} z^{-n}+\ldots\
\hline &x(n)=\alpha_{n}\
\hline Common Taylor series: & \quad \frac{1}{1-t}=\sum_{n=0}^{\infty} t^{n}, for $|t|<1\
\hline &\ln (1+t)=\sum_{n=1}^{\infty} \frac{(-1)^{n+1}}{n} t^{n}, for |t|<1\
\hline Long-division example:&\quad X(z)=\frac{1}{1+2 z^{-1}+z^{-2}} \quad 1+2 z^{-1}+z^{-2} \prod_{-1}\
\hline Partial fraction expansion: & \frac{X(z)}{z}=\sum_{i=0}^{N} \frac{a_{i}}{z-p_{i}}\
\hline & x(n)=\sum_{i=0}^{N} a_{i} p_{i}^{n} u(n)\
\hline Simple poles: & \quad a_{i}=\lim_{z \rightarrow p_{i}}\left(z-p_{i}\right) \frac{X(z)}{z}\
\hline Repeated poles: & \frac{X(z)}{z}=\ldots+\left{\frac{a_{m}}{(z-p)^{m}}+\frac{a_{m-1}}{(z-p)^{m-1}}+\ldots+\frac{a_{1}}{z-p}\right}+\ldots \
\hline & a_{m}=\lim_{z \rightarrow p} \frac{(z-p)^{m} X(z)}{z} \
\hline & a_{m-i}=\lim_{z \rightarrow p}\left{\frac{1}{i !} \frac{d^{i}}{d z^{i}}\left[\frac{(z-p)^{m} X(z)}{z}\right]\right}\
\hline Short-cut: & If X(z)=\frac{a}{1-p_{1} z^{-1}}+\frac{b}{1-p_{2} z-1} \
\hline & a=\left.\left(1-p_{1} z^{-1}\right) X(z)\right| _ { = p_{1}} \
\hline & a=\left.\left(1-p_{1} z^{-1}\right) X(z)\right| _ {z= p_{1}} \
\hline & b=\left.\left(1-p_{2} z^{-1}\right) X(z)\right| _ {z= p_{2}} \
\hline
\end{array}
$$

Correlation sequences

$$
\begin{array}{|l|c|c|}
\hline Cross-correlation& r_{x y}(l)=\sum_{n=-\infty}^{\infty} x(n) y(n-l)\
\hline Auto-correlation: &r_{x x}(l)=\sum_{n=-\infty}^{\infty} x(n) x(n-l) \
\hline Properties:&r_{x y}(l)=x(l) * y(-l) \
\hline r_{x x}(l)=r_{x x}(-l)& r_{x y}(l)=r_{y x}(-l)\
\hline\left|r_{x x}(l)\right| \leq r_{x x}(0) & \left|r_{x y}(l)\right| \leq \sqrt{r_{x x}(0) r_{y y}(0)}\
\hline Covariance:&\quad C_{x y}(l)=r_{x y}(l)-\mu_{x} \mu_{y} \
\hline Auto-covariance:& C_{x x}(l)=r_{x x}(l)-\mu_{x}^{2}\
\hline LTI input-output sequences:&r_{y y}(l)=r_{x y}(l) * h(l) \
\hline & r_{y x}(l)=r_{x x}(l) * h(l)\
\hline & r_{y y}(l)=r_{x x}(l) * r_{h h}(l)\
\hline Signal energy: & \quad E=r_{x x}(0)=\sum_{n=-\infty}^{\infty}|x(n)|^{2} \
\hline Signal power: & \quad P=\lim_{L \rightarrow \infty} \frac{1}{2 L+1} \sum_{n=-L}^{L}|x(n)|^{2}\
\hline Signal detection: & x(n)=\alpha s(n-D)+\omega(n), \quad 0 \leq n \leq L-1 \
\hline &r_{x s}(l)=\alpha r_{s s}(l-D) \
\hline Matched filter: & h(n)=s(L-1-n) ; \quad 0 \leq n \leq L-1\
\hline &y(L-1)=r_{x s}(0) \
\hline
\end{array}
$$
